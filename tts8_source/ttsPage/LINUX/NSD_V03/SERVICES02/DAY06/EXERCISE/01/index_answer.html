3 配置rsync服务器

将目录/common设置为共享，共享名为tools，只允许用户tom只读访问，对应的访问密码为12345。

参考答案

1）配置rsync服务器

[root@svr5 ~]# vim  /etc/rsyncd_users.db
tom:12345 										//每行一条用户记录
[root@svr5 ~]# chmod  600  /etc/rsyncd_users.db
[root@svr5 ~]# vim  /etc/rsyncd.conf
[tools]
    path = /common
    comment = Rsync Share Test
    read only = yes
    dont compress = *.gz *.bz2 *.tgz *.zip
    auth users = tom  								//许可的用户
    secrets file = /etc/rsyncd_users.db
[root@svr5 ~]# rsync --daemon

2）客户端访问测试

[root@svr5 ~]# rsync  ruser@192.168.4.5::tools
