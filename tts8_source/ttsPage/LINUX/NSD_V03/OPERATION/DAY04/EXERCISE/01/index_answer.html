1 如何优化提升Nginx并发数量

参考答案

[root@nginx ~]# vim /usr/local/nginx/nginx.conf
.. ..
worker_rlimit_nofile 60000;			//最大打开文件数量
events {
worker_connections 10000;				//每个worker最大并发连接数
use epoll;
}
[root@nginx ~]# vim /etc/security/limits.conf
.. ..
* 			soft    nofile  100000
* 			hard    nofile  100000

2 如何使用ab对Web服务器进行压力测试

要求：并发数为1024，总请求数为2048，测试页面为http://www.tarena.com/

参考答案

[root@localhost ~]# ab -c 2048 –n 1024 http://www.tarena.com/

3 使用Nginx如何自定义404错误页面

参考答案

[root@nginx ~]# vim /usr/local/nginx/conf/nginx.conf
.. ..
http {
fastcgi_intercept_errors on; 		//错误页面重定向
server {
error_page   404  /40x.html;		//自定义404错误页面
        location = /40x.html {
            root   html;
        }
error_page   500 502 503 504  /50x.html;
        location = /50x.html {
            root   html;
        }
}
}
